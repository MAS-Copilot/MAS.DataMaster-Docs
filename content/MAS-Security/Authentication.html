索引优化

用户索引（User）

SQL 语句：

-- 角色字段：常用于查询权限、筛选角色
CREATE INDEX `IX_User_Role` ON `User` (`Role`);

-- 创建时间：注册用户时间排序筛选
CREATE INDEX `IX_User_CreatedAt` ON `User` (`CreatedAt`);

-- 最后登录时间：用于排序或查最近活跃用户
CREATE INDEX `IX_User_LastLoginTime` ON `User` (`LastLoginTime`);

-- 登录次数：排行榜、统计分析
CREATE INDEX `IX_User_LoginCount` ON `User` (`LoginCount`);

特性标记：

[Index("IX_User_Role", "Role")]
[Index("IX_User_CreatedAt", "CreatedAt")]
[Index("IX_User_LastLoginTime", "LastLoginTime")]
[Index("IX_User_LoginCount", "LoginCount")]

操作日志索引（OperationLog）

SQL 语句：

-- 操作者 ID（外键，频繁用于筛选）
CREATE INDEX `IX_OperationLog_Username` ON `OperationLog` (`Username`);

-- 创建时间：用于时间段过滤，时间倒序分页等
CREATE INDEX `IX_OperationLog_CreatedAt` ON `OperationLog` (`CreatedAt`);

-- 操作名称：某类操作日志筛选
CREATE INDEX `IX_OperationLog_Action` ON `OperationLog` (`Action`);

-- 联合索引：用户在某时间段内的某类操作
CREATE INDEX `IX_OperationLog_Username_CreatedAt` ON `OperationLog` (`Username`, `CreatedAt`);

-- 如果资源操作类型分析比较频繁
CREATE INDEX `IX_OperationLog_ResourceId_Action` ON `OperationLog` (`ResourceId`, `Action`);

特性标记：

[Index("IX_OperationLog_Username", "Username")]
[Index("IX_OperationLog_CreatedAt", "CreatedAt")]
[Index("IX_OperationLog_Action", "Action")]
[Index("IX_OperationLog_UserId_CreatedAt", "Username", "CreatedAt")]
[Index("IX_OperationLog_ResourceId_Action", "ResourceId", "Action")]
